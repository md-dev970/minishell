enum token_type {
        CMD,
        PIPE,
        GT,
        LT,
        DGT,
        DLT,
        ARG,
        OPT
};

struct token {
        enum token_type t;
        char *value;
};

(e : end)

B -> e | S
S -> IDENT.AP
P -> PIPE.SP | e
A -> IA | OA | e
I -> (IDENT | LT | DLT)A
O -> (GT | DGT)A
